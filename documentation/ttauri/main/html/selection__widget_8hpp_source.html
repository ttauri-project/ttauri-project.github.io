112  [[nodiscard]] void update_layout(hires_utc_clock::time_point display_time_point, bool need_layout) noexcept override
113  {
114  tt_axiom(gui_system_mutex.recurse_lock_count());
115 
116  need_layout |= std::exchange(_request_relayout, false);
117 
118  if (_selecting) {
119  if (need_layout) {
120  // The overlay itself will make sure the overlay fits the window, so we give the preferred size and position
121  // from the point of view of the selection widget.
122 
123  // The overlay should start on the same left edge as the selection box and the same width.
124  // The height of the overlay should be the maximum height, which will show all the options.
125 
126  ttlet overlay_width =
127  clamp(rectangle().width() - theme::global->smallSize, _overlay_widget->preferred_size().width());
128  ttlet overlay_height = _overlay_widget->preferred_size().maximum().height();
129  ttlet overlay_x = _window_rectangle.x() + theme::global->smallSize;
130  ttlet overlay_y = std::round(_window_rectangle.middle() - overlay_height * 0.5f);
131  ttlet overlay_rectangle = aarect{overlay_x, overlay_y, overlay_width, overlay_height};
132 
133  _overlay_widget->set_layout_parameters(overlay_rectangle, overlay_rectangle);
134  }
135  }
136 
137  if (need_layout) {
138  _left_box_rectangle = aarect{0.0f, 0.0f, theme::global->smallSize, rectangle().height()};
139  _chevrons_glyph = to_font_glyph_ids(elusive_icon::ChevronUp);
140  ttlet chevrons_glyph_bbox = pipeline_SDF::device_shared::getBoundingBox(_chevrons_glyph);
141  _chevrons_rectangle =
142  align(_left_box_rectangle, scale(chevrons_glyph_bbox, theme::global->small_icon_size), alignment::middle_center);
143  _chevrons_rectangle =
144  align(_left_box_rectangle, scale(chevrons_glyph_bbox, theme::global->small_icon_size), alignment::middle_center);
145 
146  // The unknown_label is located to the right of the selection box icon.
147  _option_rectangle = aarect{
148  _left_box_rectangle.right() + theme::global->margin,

175  {
176  ttlet lock = std::scoped_lock(gui_system_mutex);

191  {
192  ttlet lock = std::scoped_lock(gui_system_mutex);

339  void repopulate_options() noexcept
340  {
341  ttlet lock = std::scoped_lock(gui_system_mutex);
342  auto option_list_ = *option_list;
343 
344  // If any of the options has a an icon, all of the options should show the icon.
345  auto show_icon = false;
346  for (ttlet & [ tag, label ] : option_list_) {
347  show_icon |= label.has_icon();
348  }
349 
350  _column_widget->clear();
351  _menu_item_widgets.clear();
352  _menu_item_callbacks.clear();
353  for (ttlet & [ tag, text ] : option_list_) {
354  auto menu_item = _column_widget->make_widget<menu_item_widget<value_type>>(tag, this->value);
355  menu_item->set_show_check_mark(true);
356  menu_item->set_show_icon(show_icon);
357  menu_item->label = text;
358 
359  _menu_item_callbacks.push_back(menu_item->subscribe([this, tag] {
360  this->value = tag;
361  this->_selecting = false;
362  }));
363 
364  _menu_item_widgets.push_back(std::move(menu_item));
365  }
366 
367  _max_option_label_height = 0.0f;
368  for (ttlet & [ tag, text ] : *option_list) {
369  _max_option_label_height = std::max(
370  _max_option_label_height,
371  stencil::make_unique(alignment::middle_left, text, theme::global->labelStyle)->preferred_extent().height());
372  }
373  }
