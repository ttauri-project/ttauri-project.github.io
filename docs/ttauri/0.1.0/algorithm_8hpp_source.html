189 template<typename InputIt1, typename InputIt2, typename BinaryPredicate>
190 inline std::pair<InputIt1,InputIt2> rmismatch(InputIt1 first1, InputIt1 last1, InputIt2 first2, InputIt2 last2, BinaryPredicate predicate) noexcept
191 {
192  auto i1 = last1;
193  auto i2 = last2;
194 
195  while (true) {
196  if (i1 == first1 && i2 == first2) {
197  return {last1, last2};

203

206  }
207  }
